a[href$="/"]:after, a[target="_blank"]:after{
    content:"ページ";
    color: #ff0000;
}
a[href$="jpg"]:after{
content:"画像";
    color: #0000ff;
}
.demo a:after	{
    font-size:83%;
    margin-left: 10px;
    padding: 5px;
    border: solid 1px;
}
